```python
"""Aggregate execution — COUNT / SUM / AVG / MIN / MAX with optional GROUP BY.

The aggregate plan sits BETWEEN Filter and Project in the SELECT pipeline.
It consumes rows from the upstream plan and emits one row per group (or
one row total when no GROUP BY is given).
"""

from __future__ import annotations
from dataclasses import dataclass
from typing import Any, Callable, Iterator, Sequence, Tuple
if TYPE_CHECKING:
    from tinydb.executor.executor import Executor
Row = tuple

# --- column-name resolver ----------------------------------------------

def _resolve_col_value(column: str, row: tuple, n2i: dict[str, int]) -> Any:
    """Return row[column_idx], or ``None`` for COUNT(*).
    Wildcard COUNT doesn't read a column — handled at the call site."""
    if column == "*":
        return None  # COUNT(*) reads no column
    if column not in n2i:
        raise UnknownColumnError(...)
    return row[n2i[column]]


# --- single-row accumulator --------------------------------------------

def _apply_agg(state, value, func: str) -> None:
    """Mutate ``state`` (a small dict) with one row's value for ``func``.

    Semantics:
    - COUNT(*) and COUNT(col): count rows; COUNT(col) skips NULL.
    - SUM(col): ignore NULL; None if no rows.
    - AVG(col): ignore NULL; None if no rows.  Implemented as (sum, count).
    - MIN(col) / MAX(col): skip NULL; tie-break on TypeError → raise.
    """
    if func == "COUNT":
        if state["count"] is None:
            state["count"] = 0
        state["count"] += 1
        return
    if value is None:
        return  # NULL ignores
    if func == "SUM":
        state["sum"] = (state["sum"] or 0) + value
        state["count"] = (state["count"] or 0) + 1
    elif func == "AVG":
        state["sum"] = (state["sum"] or 0) + value
        state["count"] = (state["count"] or 0) + 1
    elif func == "MIN":
        state["min"] = value if state["min"] is None else min(state["min"], value)
    elif func == "MAX":
        state["max"] = value if state["max"] is None else max(state["max"], value)


def _finalize_agg(state, func: str) -> Any:
    if func == "COUNT":
        return state.get("count") or 0
    if state.get("count", 0) == 0 and func in ("SUM", "AVG"):
        return None
    if func == "SUM":
        return state["sum"]
    if func == "AVG":
        return state["sum"] / state["count"]
    if func in ("MIN", "MAX"):
        return state.get(func.lower())
    raise ValueError(f"unknown aggregate func: {func!r}")


# --- plan -------------------------------------------------------------

@dataclass(frozen=True, slots=True, kw_only=True)
class Aggregate(Plan):
    """Apply the SELECT's aggregates, optionally grouped by ``keys``.

    When ``keys`` is empty, produces one row of aggregate results.
    Otherwise produces one row per distinct combination of key values,
    in input order.

    Output row shape: ``(key_tuple..., agg_value_for_column_0, agg_value_for_column_1, ...)``.
    ``Select.aggregates`` order determines the suffix positions.
    """
    src: Plan
    aggregates: Sequence[tuple]   # from Select.aggregates
    keys: Sequence[str] = ()       # from Select.group_by
    op_name: str = "Aggregate"

    @property
    def table(self) -> str:
        return self.src.table

    def open(self, ctx: Executor) -> Iterator[Row]:
        ...
```

### `src/tinydb/executor/ops.py` (extend Plan / ops)

`Plan.table` already returns the base table for any plan; extend with an aggregate-friendly rule (Aggregate inherits from src.table, same as Filter / Sort). No change needed if Plan.table property is already `self.src.table` for wrappers.

### `src/tinydb/executor/planner.py` (modify)

Update `plan_select` so that when `stmt.aggregates` is non-empty (or `GROUP BY` is non-empty), it wraps the post-Filter plan in an `Aggregate(aggregates=..., keys=stmt.group_by)` node. **Important**: the Aggregate already runs over the Filter output, so it sits between Filter and Project (before Sort/Limit).

Note: `Project` after Aggregate must NOT try to lookup aggregate result columns by name (they're synthesized). Therefore when aggregates are present, the planner's `Project.items` should pass `Select.columns` unchanged (the executor's Project will evaluate them against the aggregate's output row, where the tuple position already matches).

> Pragmatic detail: the brief's prior version of Project takes `items` which is a parallel array of source labels. Aggregates need to know they should NOT be re-evaluated by Project on the post-aggregate row — instead, the post-aggregate row layout already contains aggregate results at the SUFFIX. Easiest v0.1: when aggregates are present, Project becomes identity (yields each row). Tests below cover this.

11. Aggregate with NULL column values: `SUM(NULL)` → still None.
12. Multiple group keys: `SELECT dept, role, COUNT(*) FROM users GROUP BY dept, role` → one row per (dept, role) combo.
13. End-to-end: build index → insert some rows → aggregate → result reflects inserts.
