### New helper in `src/tinydb/types/codec.py` (add, do NOT modify existing functions)
Append at the END of the file (after `_check_int_range`):
```python
def encode_row(values: Sequence[Any], tags: Sequence["TypeTag"]) -> bytes:
    """Pack a row as a sequence of length-prefixed values.

    Each value is encoded with ``encode_value(value, tags[i])``. Used by
    the executor when a Heap stores a tuple of column-major encoded
    blobs (REQ-QEX-1).
    """
    if len(values) != len(tags):
        raise ValueError(
            f"encode_row: {len(values)} values vs {len(tags)} tags"
        )
    out = bytearray()
    for v, t in zip(values, tags):
        out += encode_value(v, t)
    return bytes(out)


def decode_row(blob: bytes, tags: Sequence["TypeTag"]) -> tuple:
    """Reverse of ``encode_row`` — returns a tuple of Python values.

    Raises if ``blob`` ends mid-value (defensive: Heap should not produce
    truncated records, but we want a clear error if it does).
    """
    out: list[Any] = []
    offset = 0
    for t in tags:
        v, offset = decode_value(blob, offset)
        out.append(v)
    if offset != len(blob):
        # not fatal — extra bytes tolerated; but flag if there's leftover
        # by raising so we don't silently mis-frame on corrupt pages
        raise ValueError(
            f"decode_row: {len(blob) - offset} trailing bytes "
            f"(expected exactly {len(blob)} consumed)"
        )
    return tuple(out)
```

### New: `src/tinydb/executor/row_iter.py`
```python
from __future__ import annotations
from typing import Iterator, Sequence
from tinydb.storage.heap import Heap, Rid
from tinydb.storage.catalog import Catalog, TableMeta
from tinydb.types.system import TypeTag
from tinydb.types.codec import decode_row

class RowMaterialised(list):
    """list[tuple] with a sibling .rids list — keeps alignment with
    Heap.scan() ordering. Used by Filter to push down row → tuple
    while keeping Rid for downstream DML hooks (T-5.5)."""

class TableScan:
    """Iterator over a Heap table — yields (rid, row_tuple) pairs."""
    def __init__(self, heap: Heap, meta: TableMeta) -> None:
        self._heap = heap
        self._meta = meta
        self._tags = tuple(c.tag for c in meta.columns)

    def __iter__(self) -> Iterator[tuple[Rid, tuple]]:
        for rid in self._heap.scan():
            blob = self._heap.read(rid)
            if blob is None:
                continue  # tombstoned slot
            yield rid, decode_row(blob, self._tags)
```

Important: `Heap.scan()` already filters tombstones itself, but the `read()` could still return `None` defensively (per Batch 2 contract). The `if blob is None: continue` is correct.

class SeqScan(Plan):

RECOMMENDED: every leaf plan carries `table: str` (already true for `SeqScan`); wrappers inherit via traversal:

```python
@dataclass(frozen=True, slots=True)
class Plan:
    op_name: str
    @property
    def table(self) -> str:
        """The base table this plan reads/writes. Wrappers traverse
        to the leaf. Override in leaf plan subclasses (SeqScan)."""
        raise NotImplementedError
```

`SeqScan.table` returns `self.table`; `Filter.table` returns `self.src.table`; etc.
